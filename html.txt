  <div class="subtitle">Polling every 3 seconds for new messages</div>
  <div class="message-list" id="messageList">
    <div class="empty">Waiting for messages…</div>
  </div>

<script>
  const listEl = document.getElementById('messageList');
  let lastRenderedCount = -1;

  function formatTime(iso) {
    const d = new Date(iso);
    return d.toLocaleString();
  }

  function render(messages) {
    if (messages.length === lastRenderedCount) return; // avoid needless re-render
    lastRenderedCount = messages.length;

    if (messages.length === 0) {
      listEl.innerHTML = '<div class="empty">Waiting for messages…</div>';
      return;
    }

    listEl.innerHTML = messages.map(m => `
      <div class="message-card">
        <div class="message-meta">
          <span class="message-from">${escapeHtml(m.from)}</span>
          <span>${formatTime(m.receivedAt)}</span>
        </div>
        <div class="message-body">${escapeHtml(m.body)}</div>
      </div>
    `).join('');
  }

  function escapeHtml(str) {
    const div = document.createElement('div');
    div.textContent = str ?? '';
    return div.innerHTML;
  }

  async function fetchMessages() {
    try {
      const res = await fetch('/api/messages');
      const data = await res.json();
      render(data);
    } catch (err) {
      console.error('Failed to fetch messages', err);
    }
  }

  fetchMessages();
  setInterval(fetchMessages, 3000);
</script>
